pub fn load_poscar<R>(reader: R) -> System where
    R: BufRead, 

Returns a System object initialized from POSCAR formatted data.

Arguments

  • reader - File object or buffer to read from.

Examples

use velvet_core::system::System;
use velvet_convert::load_poscar;

let system = load_poscar("\
    Cubic BN
    3.57
    0.0 0.5 0.5
    0.5 0.0 0.5
    0.5 0.5 0.0
    B N
    1 1
    Direct
    0.00 0.00 0.00
    0.25 0.25 0.25
    ".as_bytes());

assert_eq!(system.size(), 2);
